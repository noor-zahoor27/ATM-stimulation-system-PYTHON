#ATM BALANCE CHECK
Pin=201918
Balance=1000
max_attempts=3
attempts=0
while attempts < max_attempts:
    entered_pin = input("Enter your PIN: ")
    if entered_pin == pin:
        print("Login successful!\n")
        break
    else:
        attempts += 1
        print(f"Incorrect PIN! Attempts left: {max_attempts - attempts}")
else:
    print("Too many incorrect attempts. Exiting.")
    exit()
while True:
    print("\n-------- ATM Menu --------")
    print("1. Check Balance")
    print("2. Withdraw Money")
    print("3. Deposit Money")
    print("4. Change PIN")
    print("5. Transfer Money")
    print("6. Exit")
    choice = input("Select an option (1-6): ")

    if choice == "1":
        print(f"Your current balance is: Rs {balance}")

    elif choice == "2":
        amount = int(input("Enter amount to withdraw: "))
        if amount <= balance:
            balance -= amount
            print(f"Rs {amount} withdrawn. New balance: Rs {balance}")
        else:
            print("Insufficient balance!")

    elif choice == "3":
        amount = int(input("Enter amount to deposit: "))
        balance += amount
        print(f"Rs {amount} deposited. New balance: Rs {balance}")
    elif choice == "4":
        new_pin = input("Enter new PIN: ")
        confirm_pin = input("Confirm new PIN: ")
        if new_pin == confirm_pin:
            pin = new_pin
            print("PIN changed successfully.")
        else:
            print("PINs do not match!")
    elif choice == "5":
        recipient = input("Enter recipient account number: ")
        amount = int(input("Enter amount to transfer: "))
        if amount <= balance:
            balance -= amount
            print(f"Rs {amount} transferred to account {recipient}. New balance: Rs {balance}")
        else:
            print("Insufficient balance!")

    elif choice == "6":
        print("Thank you for using the ATM. Goodbye!")
        break

    else:
        print("Invalid option. Please choose between 1 and 6.")
